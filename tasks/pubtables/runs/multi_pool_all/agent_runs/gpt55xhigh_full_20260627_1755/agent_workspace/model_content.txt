def normalize_bbox(raw):
    if raw is None:
        return None

    if isinstance(raw, dict):
        keys = {str(k).lower(): v for k, v in raw.items()}
        if all(k in keys for k in ("x0", "y0", "x1", "y1")):
            return [float(keys["x0"]), float(keys["y0"]), float(keys["x1"]), float(keys["y1"])]
        if all(k in keys for k in ("left", "top", "right", "bottom")):
            return [float(keys["left"]), float(keys["top"]), float(keys["right"]), float(keys["bottom"])]
        if all(k in keys for k in ("xmin", "ymin", "xmax", "ymax")):
            return [float(keys["xmin"]), float(keys["ymin"]), float(keys["xmax"]), float(keys["ymax"])]
        if all(k in keys for k in ("x", "y", "w", "h")):
            x, y, w, h = float(keys["x"]), float(keys["y"]), float(keys["w"]), float(keys["h"])
            return [x, y, x + w, y + h]
        if all(k in keys for k in ("x", "y", "width", "height")):
            x, y = float(keys["x"]), float(keys["y"])
            w, h = float(keys["width"]), float(keys["height"])
            return [x, y, x + w, y + h]

    if isinstance(raw, (list, tuple)) and len(raw) >= 4:
        vals = [float(v) for v in raw[:4]]
        x0, y0, x1, y1 = vals
        if x1 < x0:
            x0, x1 = x1, x0
        if y1 < y0:
            y0, y1 = y1, y0
        return [x0, y0, x1, y1]

    return None


def bbox_area(b):
    if not b:
        return 0.0
    return max(0.0, b[2] - b[0]) * max(0.0, b[3] - b[1])


def bbox_overlap(a, b):
    if not a or not b:
        return 0.0
    x0 = max(a[0], b[0])
    y0 = max(a[1], b[1])
    x1 = min(a[2], b[2])
    y1 = min(a[3], b[3])
    return max(0.0, x1 - x0) * max(0.0, y1 - y0)


def bbox_union(boxes):
    boxes = [b for b in boxes if b]
    if not boxes:
        return None
    return [
        min(b[0] for b in boxes),
        min(b[1] for b in boxes),
        max(b[2] for b in boxes),
        max(b[3] for b in boxes),
    ]


def center(b):
    return ((b[0] + b[2]) / 2.0, (b[1] + b[3]) / 2.0)


def text_from_obj(obj):
    for key in ("text", "value", "word", "token", "content", "ocr_text"):
        if isinstance(obj, dict) and key in obj and obj[key] is not None:
            return str(obj[key])
    return None


def bbox_from_obj(obj):
    if not isinstance(obj, dict):
        return None
    for key in ("bbox", "bounding_box", "box", "bounds", "rect"):
        if key in obj:
            b = normalize_bbox(obj[key])
            if b:
                return b
    b = normalize_bbox(obj)
    return b


def walk_json(obj, path=""):
    yield path, obj
    if isinstance(obj, dict):
        for k, v in obj.items():
            yield from walk_json(v, f"{path}.{k}" if path else str(k))
    elif isinstance(obj, list):
        for i, v in enumerate(obj):
            yield from walk_json(v, f"{path}[{i}]")


def extract_words(data):
    words = []
    seen = set()

    for path, obj in walk_json(data):
        if not isinstance(obj, dict):
            continue
        txt = text_from_obj(obj)
        box = bbox_from_obj(obj)
        if txt is None or box is None:
            continue
        txt = txt.strip()
        if not txt:
            continue
        key = (round(box[0], 3), round(box[1], 3), round(box[2], 3), round(box[3], 3), txt)
        if key in seen:
            continue
        seen.add(key)
        words.append({"text": txt, "bbox": box, "source_path": path})

    return words


def likely_table_bbox_key(path):
    lower = path.lower()
    if "table" not in lower:
        return False
    if any(term in lower for term in ("word", "token", "cell", "caption", "footnote", "page")):
        return False
    return any(term in lower for term in ("bbox", "bounding", "box", "bounds", "rect"))


def extract_table_bbox(data, words):
    candidates = []

    for path, obj in walk_json(data):
        b = None
        if isinstance(obj, dict):
            for key in ("table_bbox", "table_bounding_box", "table_box", "bbox", "bounding_box", "bounds", "rect"):
                if key in obj:
                    candidate = normalize_bbox(obj[key])
                    if candidate:
                        joined = f"{path}.{key}".lower()
                        if "table" in joined or key.startswith("table_"):
                            b = candidate
                            break
            if b is None:
                candidate = normalize_bbox(obj)
                if candidate and likely_table_bbox_key(path):
                    b = candidate
        elif likely_table_bbox_key(path):
            b = normalize_bbox(obj)

        if b and bbox_area(b) > 0:
            area = bbox_area(b)
            word_centers_inside = 0
            for w in words:
                cx, cy = center(w["bbox"])
                if b[0] <= cx <= b[2] and b[1] <= cy <= b[3]:
                    word_centers_inside += 1
            candidates.append((word_centers_inside, area, path, b))

    if candidates:
        candidates.sort(key=lambda item: (item[0], -item[1]), reverse=True)
        return candidates[0][3], f"provided:{candidates[0][2]}"

    if words:
        return bbox_union([w["bbox"] for w in words]), "inferred:all_words_union"

    return None, "missing"


def filter_words_to_table(words, table_bbox):
    kept = []
    excluded = []

    for w in words:
        b = w["bbox"]
        cx, cy = center(b)
        inside_center = table_bbox and table_bbox[0] <= cx <= table_bbox[2] and table_bbox[1] <= cy <= table_bbox[3]
        overlap_ratio = bbox_overlap(b, table_bbox) / max(1e-9, bbox_area(b)) if table_bbox else 1.0

        if inside_center or overlap_ratio >= 0.65:
            kept.append(w)
        else:
            excluded.append(w)

    return kept, excluded


def median(values, default=0.0):
    values = [v for v in values if v is not None and not math.isnan(v)]
    return statistics.median(values) if values else default


def group_rows(words):
    if not words:
        return []

    heights = [w["bbox"][3] - w["bbox"][1] for w in words]
    tol = max(4.0, median(heights, 8.0) * 0.65)
    sorted_words = sorted(words, key=lambda w: (center(w["bbox"])[1], center(w["bbox"])[0]))

    rows = []
    for w in sorted_words:
        cy = center(w["bbox"])[1]
        best_idx = None
        best_dist = None

        for idx, row in enumerate(rows):
            dist = abs(cy - row["cy"])
            if dist <= tol and (best_dist is None or dist < best_dist):
                best_idx = idx
                best_dist = dist

        if best_idx is None:
            rows.append({"cy": cy, "words": [w]})
        else:
            row = rows[best_idx]
            row["words"].append(w)
            row["cy"] = sum(center(x["bbox"])[1] for x in row["words"]) / len(row["words"])

    rows.sort(key=lambda r: r["cy"])
    for row in rows:
        row["words"].sort(key=lambda w: center(w["bbox"])[0])
        row["bbox"] = bbox_union([w["bbox"] for w in row["words"]])

    return rows


def numeric_like(text):
    s = text.strip()
    return bool(re.fullmatch(r"[-+]?(\d+(\.\d+)?|\.\d+)%?", s))


def body_score(row):
    texts = [w["text"] for w in row["words"]]
    if not texts:
        return 0.0
    numeric_count = sum(1 for t in texts if numeric_like(t))
    digit_count = sum(1 for t in texts if re.search(r"\d", t))
    return (numeric_count * 2 + digit_count) / len(texts)


def infer_header_row_count(rows):
    if not rows:
        return 0

    for idx, row in enumerate(rows):
        if idx == 0:
            continue
        if body_score(row) >= 0.25:
            return idx

    if len(rows) >= 2:
        return 1
    return 0


def find_large_gaps(row):
    words = row["words"]
    if len(words) < 2:
        return []
    widths = [w["bbox"][2] - w["bbox"][0] for w in words]
    med_width = median(widths, 10.0)
    gaps = []
    for left, right in zip(words, words[1:]):
        gap = right["bbox"][0] - left["bbox"][2]
        if gap > max(10.0, med_width * 0.9):
            gaps.append((left["bbox"][2], right["bbox"][0], gap))
    return gaps


def infer_column_bands(rows, table_bbox):
    body_start = infer_header_row_count(rows)
    candidate_rows = rows[body_start:] if body_start < len(rows) else rows
    if not candidate_rows:
        candidate_rows = rows

    max_cells = max((len(find_large_gaps(r)) + 1 for r in candidate_rows), default=1)
    chosen_rows = [r for r in candidate_rows if len(find_large_gaps(r)) + 1 == max_cells]
    boundary_votes = []

    for row in chosen_rows:
        for x0, x1, _gap in find_large_gaps(row):
            boundary_votes.append((x0 + x1) / 2.0)

    if boundary_votes:
        boundary_votes.sort()
        widths = [w["bbox"][2] - w["bbox"][0] for r in rows for w in r["words"]]
        tol = max(8.0, median(widths, 8.0) * 0.75)
        clusters = []
        for x in boundary_votes:
            if not clusters or abs(x - clusters[-1]["x"]) > tol:
                clusters.append({"x": x, "items": [x]})
            else:
                clusters[-1]["items"].append(x)
                clusters[-1]["x"] = sum(clusters[-1]["items"]) / len(clusters[-1]["items"])

        boundaries = [c["x"] for c in clusters]
    else:
        # Fallback: cluster word left edges from body rows.
        lefts = sorted(w["bbox"][0] for r in candidate_rows for w in r["words"])
        if not lefts:
            boundaries = []
        else:
            tol = max(12.0, median([w["bbox"][2] - w["bbox"][0] for r in rows for w in r["words"]], 10.0))
            clusters = []
            for x in lefts:
                if not clusters or abs(x - clusters[-1]["x"]) > tol:
                    clusters.append({"x": x, "items": [x]})
                else:
                    clusters[-1]["items"].append(x)
                    clusters[-1]["x"] = sum(clusters[-1]["items"]) / len(clusters[-1]["items"])
            starts = [c["x"] for c in clusters]
            boundaries = [(a + b) / 2.0 for a, b in zip(starts, starts[1:])]

    left = table_bbox[0] if table_bbox else min(w["bbox"][0] for r in rows for w in r["words"])
    right = table_bbox[2] if table_bbox else max(w["bbox"][2] for r in rows for w in r["words"])

    clean_boundaries = []
    for b in boundaries:
        if left < b < right and (not clean_boundaries or abs(b - clean_boundaries[-1]) > 3.0):
            clean_boundaries.append(b)

    edges = [left] + clean_boundaries + [right]
    bands = []
    for idx, (x0, x1) in enumerate(zip(edges, edges[1:])):
        bands.append({"col_id": idx, "x0": x0, "x1": x1, "xc": (x0 + x1) / 2.0})

    return bands


def assign_col(word, bands):
    b = word["bbox"]
    cx, _ = center(b)
    for band in bands:
        if band["x0"] <= cx <= band["x1"]:
            return band["col_id"]

    best = None
    best_overlap = -1.0
    for band in bands:
        overlap = max(0.0, min(b[2], band["x1"]) - max(b[0], band["x0"]))
        if overlap > best_overlap:
            best = band["col_id"]
            best_overlap = overlap

    if best is not None:
        return best

    return 0


def join_tokens(tokens):
    out = ""
    for token in tokens:
        if not out:
            out = token
        elif re.match(r"^[,.;:%)]$", token):
            out += token
        elif out.endswith("("):
            out += token
        elif token in {"/", "-"} or out.endswith(("/", "-")):
            out += token
        else:
            out += " " + token
    return out.strip()


def reconstruct_cells(rows, bands):
    header_count = infer_header_row_count(rows)
    cells = []

    for row_id, row in enumerate(rows):
        buckets = {}
        for w in row["words"]:
            col = assign_col(w, bands)
            buckets.setdefault(col, []).append(w)

        for col_id in sorted(buckets):
            cell_words = sorted(buckets[col_id], key=lambda w: w["bbox"][0])
            text = join_tokens([w["text"] for w in cell_words])
            if not text:
                continue
            box = bbox_union([w["bbox"] for w in cell_words])
            cells.append({
                "row_id": row_id,
                "col_id": col_id,
                "row_span": 1,
                "col_span": infer_col_span(box, bands),
                "is_header": row_id < header_count,
                "text": text,
                "_bbox": box,
            })

    infer_row_spans(cells, rows, header_count)
    return cells, header_count


def infer_col_span(box, bands):
    if not box or not bands:
        return 1

    covered = []
    for band in bands:
        overlap = max(0.0, min(box[2], band["x1"]) - max(box[0], band["x0"]))
        band_width = max(1e-9, band["x1"] - band["x0"])
        if overlap / band_width >= 0.45:
            covered.append(band["col_id"])

    return max(1, len(covered))


def infer_row_spans(cells, rows, header_count):
    if header_count <= 1:
        return

    cells_by_pos = {(c["row_id"], c["col_id"]): c for c in cells}

    for c in cells:
        if not c["is_header"]:
            continue
        if c["row_id"] >= header_count - 1:
            continue

        has_below = any((r, c["col_id"]) in cells_by_pos for r in range(c["row_id"] + 1, header_count))
        if not has_below:
            c["row_span"] = header_count - c["row_id"]


def normalize_label(text):
    return re.sub(r"[^a-z0-9]+", "", text.lower())


def infer_column_roles(cells, bands, header_count):
    roles = {}
    header_by_col = {i: [] for i in range(len(bands))}

    for c in cells:
        if c["is_header"]:
            for col in range(c["col_id"], min(len(bands), c["col_id"] + c["col_span"])):
                header_by_col.setdefault(col, []).append(c["text"])

    for col_id in range(len(bands)):
        header_text = " ".join(header_by_col.get(col_id, []))
        norm = normalize_label(header_text)
        if "method" in norm or "model" in norm or "approach" in norm or "system" in norm:
            roles[col_id] = "method"
        elif "dataset" in norm or "data" in norm or "benchmark" in norm or "corpus" in norm:
            roles[col_id] = "dataset"
        elif "accuracy" in norm or norm in {"acc"} or "acc" in norm:
            roles[col_id] = "accuracy"
        elif norm == "f1" or "f1score" in norm or "fmeasure" in norm or "macrof1" in norm or "microf1" in norm:
            roles[col_id] = "f1"
        elif "note" in norm or "remark" in norm or "comment" in norm:
            roles[col_id] = "notes"

    unassigned = [i for i in range(len(bands)) if i not in roles]
    preferred = ["method", "dataset", "accuracy", "f1", "notes"]
    for role, col_id in zip([r for r in preferred if r not in roles.values()], unassigned):
        roles[col_id] = role

    return roles


def parse_number(value):
    if value is None:
        return None
    s = str(value).strip()
    if not s:
        return None

    match = re.search(r"[-+]?(?:\d+(?:\.\d+)?|\.\d+)", s.replace(",", ""))
    if not match:
        return None

    try:
        return float(match.group(0))
    except ValueError:
        return None


def format_number(value):
    n = parse_number(value)
    if n is None:
        return str(value).strip() if value is not None else ""
    text = f"{n:.6f}".rstrip("0").rstrip(".")
    return text


def normalize_metrics(cells, bands, header_count):
    roles = infer_column_roles(cells, bands, header_count)
    body_rows = sorted(set(c["row_id"] for c in cells if not c["is_header"]))
    metrics = []

    for row_id in body_rows:
        row_cells = [c for c in cells if c["row_id"] == row_id and not c["is_header"]]
        row = {field: "" for field in REQUIRED_METRIC_FIELDS}

        for c in row_cells:
            role = roles.get(c["col_id"])
            if role in row:
                existing = row[role]
                row[role] = (existing + " " + c["text"]).strip() if existing else c["text"]

        if not any(row.values()):
            continue

        row["accuracy"] = format_number(row.get("accuracy", ""))
        row["f1"] = format_number(row.get("f1", ""))

        if row["method"] or row["dataset"] or row["accuracy"] or row["f1"]:
            metrics.append(row)

    return metrics, roles


def audit_metrics(metrics, cells, excluded_words, table_bbox_source, roles, bands):
    issues = []

    for field in REQUIRED_METRIC_FIELDS:
        if field not in REQUIRED_METRIC_FIELDS:
            issues.append({"category": "missing_required_column", "detail": f"Missing required column: {field}"})

    for idx, row in enumerate(metrics):
        for field in ("method", "dataset"):
            if not str(row.get(field, "")).strip():
                issues.append({
                    "category": "empty_required_text",
                    "row_index": idx,
                    "field": field,
                    "detail": f"Empty required text field {field}",
                })

        for field in ("accuracy", "f1"):
            if parse_number(row.get(field)) is None:
                issues.append({
                    "category": "non_numeric_metric",
                    "row_index": idx,
                    "field": field,
                    "value": row.get(field, ""),
                    "detail": f"Metric field {field} is not numeric",
                })

    seen = {}
    for idx, row in enumerate(metrics):
        key = (
            str(row.get("method", "")).strip().lower(),
            str(row.get("dataset", "")).strip().lower(),
        )
        if key in seen:
            issues.append({
                "category": "duplicate_record",
                "row_index": idx,
                "first_row_index": seen[key],
                "detail": "Duplicate method/dataset record",
            })
        else:
            seen[key] = idx

    if excluded_words:
        issues.append({
            "category": "excluded_non_table_text",
            "count": len(excluded_words),
            "detail": "Words outside the table bounding box were excluded from normalized metric rows.",
        })

    if table_bbox_source.startswith("inferred") or table_bbox_source == "missing":
        issues.append({
            "category": "ambiguous_structure",
            "detail": f"Table bounding box was {table_bbox_source}.",
        })

    role_values = set(roles.values())
    missing_roles = [field for field in REQUIRED_METRIC_FIELDS if field not in role_values]
    if missing_roles:
        issues.append({
            "category": "ambiguous_structure",
            "detail": f"Could not map header geometry/text to roles: {', '.join(missing_roles)}",
        })

    best_by_dataset = {}
    by_dataset = {}
    for row in metrics:
        ds = str(row.get("dataset", "")).strip()
        f1 = parse_number(row.get("f1"))
        if ds and f1 is not None:
            by_dataset.setdefault(ds, []).append((f1, row))

    for dataset, candidates in sorted(by_dataset.items()):
        candidates.sort(key=lambda item: item[0], reverse=True)
        best_f1, best_row = candidates[0]
        best_by_dataset[dataset] = {
            "method": best_row.get("method", ""),
            "f1": best_f1,
            "accuracy": parse_number(best_row.get("accuracy")),
            "notes": best_row.get("notes", ""),
        }

    return {
        "row_count": len(metrics),
        "best_by_dataset": best_by_dataset,
        "issues": issues,
    }


def write_cells_csv(path, cells):
    ensure_parent(path)
    with open(path, "w", newline="", encoding="utf-8") as f:
        writer = csv.DictWriter(f, fieldnames=CELL_FIELDS)
        writer.writeheader()
        for c in cells:
            writer.writerow({
                "row_id": c["row_id"],
                "col_id": c["col_id"],
                "row_span": c["row_span"],
                "col_span": c["col_span"],
                "is_header": "true" if c["is_header"] else "false",
                "text": c["text"],
            })

def write_summary_md(path, audit):
    ensure_parent(path)

    lines = [
        "# OCR Table Extraction Summary",
        "",
        f"- Normalized metric rows: {audit['row_count']}",
    ]

    if audit["best_by_dataset"]:
        lines.append("- Best method by F1 score:")
        for dataset, best in audit["best_by_dataset"].items():
            f1 = f"{best['f1']:.6f}".rstrip("0").rstrip(".")
            accuracy = best.get("accuracy")
            acc_text = ""
            if accuracy is not None:
                acc_text = f", accuracy {f'{accuracy:.6f}'.rstrip('0').rstrip('.')}"
            lines.append(f"  - {dataset}: {best['method']} (F1 {f1}{acc_text})")
    else:
        lines.append("- Best method by F1 score: no valid dataset/F1 groups were available.")

    if audit["issues"]:
        lines.append("- Audit issues:")
        for issue in audit["issues"]:
            detail = issue.get("detail", issue.get("category", "Issue recorded"))
            lines.append(f"  - {issue.get('category', 'issue')}: {detail}")
    else:
        lines.append("- Audit issues: none found.")

    with open(path, "w", encoding="utf-8") as f:
        f.write("\n".join(lines) + "\n")

    input_path = get_env_path("ORIGINAL_WORDS_JSON", DEFAULT_ORIGINAL_WORDS_JSON)
    cells_path = get_env_path("OUTPUT_CELLS_CSV", DEFAULT_OUTPUT_CELLS_CSV)
    metrics_path = get_env_path("OUTPUT_METRICS_CSV", DEFAULT_OUTPUT_METRICS_CSV)
    audit_path = get_env_path("OUTPUT_AUDIT_JSON", DEFAULT_OUTPUT_AUDIT_JSON)
    summary_path = get_env_path("SUMMARY_MD", DEFAULT_SUMMARY_MD)

    with open(input_path, "r", encoding="utf-8") as f:
        data = json.load(f)

    all_words = extract_words(data)
    table_bbox, table_bbox_source = extract_table_bbox(data, all_words)
    table_words, excluded_words = filter_words_to_table(all_words, table_bbox)

    rows = group_rows(table_words)
    bands = infer_column_bands(rows, table_bbox)
    cells, header_count = reconstruct_cells(rows, bands)
    metrics, roles = normalize_metrics(cells, bands, header_count)

    audit = audit_metrics(
        metrics=metrics,
        cells=cells,
        excluded_words=excluded_words,
        table_bbox_source=table_bbox_source,
        roles=roles,
        bands=bands,
    )

    write_cells_csv(cells_path, cells)
    write_metrics_csv(metrics_path, metrics)
    write_audit_json(audit_path, audit)
    write_summary_md(summary_path, audit)


if __name__ == "__main__":
    main()
